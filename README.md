    <script>
        // Open Order Modal
        function openOrderModal(product, price) {
            document.getElementById('orderModal').style.display = 'block';
            document.getElementById('productName').value = product;
            document.getElementById('productPrice').value = price;
        }
        
        // Close Order Modal
        function closeOrderModal() {
            document.getElementById('orderModal').style.display = 'none';
        }
        
        // Submit Order
        function submitOrder(event) {
            event.preventDefault();
            
            // Get form values
            const product = document.getElementById('productName').value;
            const price = document.getElementById('productPrice').value;
            const name = document.getElementById('customerName').value;
            const phone = document.getElementById('customerPhone').value;
            const address = document.getElementById('customerAddress').value;
            const quantity = document.getElementById('quantity').value;
            const notes = document.getElementById('orderNotes').value;
            
            // Calculate total price
            const totalPrice = parseInt(price.replace('₹', '')) * parseInt(quantity);
            
            // Create order message for WhatsApp
            const message = `New Order Details:%0A%0AProduct: ${product}%0APrice: ${price}%0AQuantity: ${quantity}%0ATotal: ₹${totalPrice}%0A%0ACustomer: ${name}%0APhone: ${phone}%0AAddress: ${address}%0ANotes: ${notes}`;
            
            // Open WhatsApp with order details (replace with your number)
            const ownerPhone = "919813490892"; // Your number
            window.open(`https://wa.me/${ownerPhone}?text=${message}`, '_blank');
            
            // Close modal
            closeOrderModal();
            
            // Reset form
            document.getElementById('orderForm').reset();
            
            // Show success message
            alert('Order placed successfully! You will be redirected to WhatsApp to confirm.');
        }
        
        // Close modal when clicking outside
        window.onclick = function(event) {
            const modal = document.getElementById('orderModal');
            if (event.target == modal) {
                closeOrderModal();
            }
        }
    </script>
